Class EventEmitter

事件订阅器和发送器

example


import { EventEmitter } from 'gzb-jssdk'

const emitter = new EventEmitter()
emitter.addListener('foo', data => {
  console.log('foo event emitted', data)
})

emitter.emit('foo', 'bar')
// => foo event emitted bar
